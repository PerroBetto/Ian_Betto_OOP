Game Module

Contains the Game singleton manager for everything within the game.

As per the tutorial online, a game loop should look like this:

>>> while True:
>>>     event()
>>>     loop()
>>>     render()
>>>     sound()

We handle events first, then run the next game loop (next frame) and then render the game to the screen last.

 51    def __init__(self, seed: Any, resolution: tuple[int, int] | None = None) -> None:
 52        """
 53        Class Game Initializer.
 54        > Initializes variables belonging to Game.
 55
 56        > Sets mixer values before initialization
 57
 58        Args:
 59            resolution (tuple[int, int] | None, optional): Game resolution.
 60            Defaults to (1280, 720)
 61
 62        _running = false.

 77    def _game_init(self) -> None:
 78        """
 79        Game initializer
 80        > Initializes the game objects when they are wanted.
 81
 82        > Opposed to __init__, this method should be called to initialize pygame objects.
 83
 84        > Should also initialize world.
 85
 86        Things to be initialized:
 87        * pygame
 88        * screen
 89        * world
 90        * game state
 91        """
 92        pygame.init()
 93        self._screen: pygame.Surface = pygame.display.set_mode(
 94            self._resolution, pygame.HWSURFACE)
 95        pygame.mixer.init()
 96        self._running = True
 97        self._world: World = World(self._seed)
 98        self._curr_music: str = str()
 99

102    def event_handler(self) -> None:
103        """
104        Event Handler
105        > Handles all PyGame events.
106
107        Main events to handle:
108        * Quit
109            * Should close out of the game.
110        * etc. (May need other events)
111        """
112        for event in pygame.event.get():
113            if event.type == locals.QUIT:
114                self._running = False

118    def on_render(self) -> None:
119        """
120        Rendering method
121        > Draws all objects to the screen by calling all draw() methods.
122
123        > Should probably only need to call _world.draw()
124
125        Objects to be drawn:
126        * World
127            * Entity
128                * Player
129                * Enemy
130                * etc.
131            * Item
132            * etc.
133        * etc.
134
135        Also calls sound handler.
136        """
137        self._screen.fill("black")
138        items = self._world.render()
139        for i, n in enumerate(items):
140            self._screen.blit(items[i][0], items[i][1])
141        pygame.display.flip()
142
143        self.sound_handler()
144
145    # --- sound module ---
146
147    def sound_handler(self) -> None:
148        """
149        Sound Handler method.
150
151        May not be needed, but if it is:
152        > Handles all object sounds in a queue
153
154        Sound includes:
155        * Music
156        * Sound effects
157
158        STEPS:
159            1. Retrieve sound data from WORLD
160                * includes both sounds and music
161            2. Play music.
162                * If the music changed from the last, fade them.
163
164        All of the above should be done at the _VOLUME constant.
165        """
166        # set volumes
167        music.set_volume(self._VOLUME)
168
169        world_music: str = self._world.music
170
171        if world_music != self._curr_music:  # Fade out music.
172            music.queue(world_music, loops=1)
173            if pygame.mixer.music.get_busy():
174                music.fadeout(100)

188        """
189        pygame.display.quit()

197        Loop method.
198        > Calls the loop method of all in game objects with a loop method.
199
200        Those with loop methods should include:
201        * World
202            * Entity
203                * Player
204                * Enemy
205                * etc.
206            * Items
207            * etc.
208        * etc.

211        self._world.loop(delta)
212        # print(self._framerate.get_fps())

214    # --- run game module ---
215
216    def run_game(self) -> None:
217        """
218        Run Game method.
219        > Starts and continues to run the game in a while loop.
220
221        > Everything that the game does should be within the while-loop.
222
223        Game actions during run-time includes:
224        * Handling events
225        * Main game loop
226            * Player actions and movement
227            * Enemy actions and movement
228            * Dungeon room logic
229            * Puzzle logic
230            * Inventory handling
231            * etc.
232        * Render handler
233            * Dungeon rendering
234            * Player rendering
235            * Enemy rendering
236            * etc.
237            * also handles sounds!
238        """
239        self._game_init()
240        while self._running:
241            self.event_handler()
242            self.on_loop()
243            self.on_render()
244        self.cleanup()
